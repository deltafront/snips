package snips;

import java.util.List;

public class RowValues
{
    public final String campaignName;
    public final String bidStrategyName;
    public final String adGroupName;

    public RowValues(RowIndices indices, List<String> values)
    {
        this.adGroupName = values.get(indices.adGroupNameIndex);
        this.bidStrategyName = values.get(indices.bidStrategyNameIndex);
        this.campaignName = values.get(indices.campaignNameIndex);
    }
}
